## Build lib3mf

```bash
git clone https://github.com/3MFConsortium/lib3mf
cd lib3mf
git submodule update --init
bash cmake/GenerateMakeFast.sh
cd build
make
cd ..
```

## Build wheel

```bash
mkdir -p lib3mf_pkg/lib3mf
cp Autogenerated/Bindings/Python/Lib3MF.py lib3mf_pkg/lib3mf/
cp build/lib3mf.so* lib3mf_pkg/lib3mf/
echo "from .Lib3MF import *" > lib3mf_pkg/lib3mf/__init__.py

cat > lib3mf_pkg/pyproject.toml << 'EOF'
[build-system]
requires = ["setuptools>=61.0", "wheel"]
build-backend = "setuptools.build_meta"

[project]
name = "lib3mf"
version = "2.4.1"
description = "lib3mf Python bindings"

[tool.setuptools.packages.find]
where = ["."]

[tool.setuptools.package-data]
lib3mf = ["*.so"]
EOF

cd lib3mf_pkg
pip install setuptools>=75 build wheel auditwheel patchelf

python -m build -n -w
```

## Delocate wheel

```bash
auditwheel repair dist/lib3mf-2.4.1-py3-none-any.whl --plat manylinux_2_31_aarch64 -w wheelhouse/
```
